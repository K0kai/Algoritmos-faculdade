Algoritmo InverterNumero;
Var
    numero, unidade, dezena, centena, numeroInvertido: inteiro;

Inicio
    escreva("Digite um número inteiro de três dígitos: ")
    leia(numero)

    centena <- numero / 100
    dezena <- (numero % 100) / 10
    unidade <- numero % 10

    numeroInvertido <- unidade * 100 + dezena * 10 + centena

    escreva("O número invertido é: ", numeroInvertido)
FimAlgoritmo
