Algoritmo DivisaoComResto;
Var
    num1, num2, resto: real;

Inicio
    escreva("Digite o primeiro número: ")
    leia(num1)
    escreva("Digite o segundo número: ")
    leia(num2)

    se(num2 > 0) então
        resto <- num1 % num2
        escreva("\nQual é o resultado: ", num1 / num2)
        escreva("\nO resto da divisão de ", num1, " por ", num2, " é ", resto)
    senao
        escreva("\nNão é possível dividir por zero.")
    fimse
FimAlgoritmo
